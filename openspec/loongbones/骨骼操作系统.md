# 骨骼操作系统源码分析

## 1. 功能概述

骨骼操作系统是 DragonBones 动画体系中最核心的逻辑单元之一，负责实现动画中的平移、旋转、缩放等变换操作。一个骨架中可以包含多个骨骼，形成层级关系，通过骨骼的变换驱动整个骨架的动画效果。

### 核心特性
- **层级骨骼结构** - 支持父子骨骼关系，形成骨骼树
- **变换计算** - 实现平移、旋转、缩放的矩阵运算
- **动画插值** - 支持骨骼动画的平滑过渡
- **约束系统** - 集成IK、路径等约束机制
- **性能优化** - 采用缓存和脏检查机制

## 3. 核心实现

### 3.1 骨骼类实现

```typescript
/**
 * 骨骼类 - 负责动画中的平移、旋转、缩放实现
 *
 * 职责：
 * - 管理骨骼的变换状态
 * - 处理骨骼层级关系
 * - 执行动画插值计算
 * - 集成约束系统
 *
 * 依赖：
 * - TransformObject: 基础变换对象
 * - BoneData: 骨骼数据定义
 * - Constraint: 约束系统
 *
 * 性能考虑：
 * - 使用脏检查机制减少不必要的计算
 * - 缓存帧索引优化动画查找
 * - 对象池管理减少内存分配
 */
export class Bone extends TransformObject {
    // 骨骼属性
    public offsetMode: OffsetMode;                    // 偏移模式
    public readonly animationPose: Transform = new Transform(); // 动画姿势

    // 脏检查标志
    public _transformDirty: boolean;                  // 变换脏标志
    public _childrenTransformDirty: boolean;          // 子变换脏标志
    protected _localDirty: boolean;                   // 本地变换脏标志

    // 约束相关
    public _hasConstraint: boolean;                   // 是否有约束

    // 数据引用
    public _boneData: BoneData;                       // 骨骼数据
    protected _parent: Bone | null;                   // 父骨骼

    /**
     * 骨骼更新逻辑
     */
    public update(dirty: boolean): void {
        // 1. 检查是否需要更新
        if (!this._localDirty && !dirty) {
            return;
        }

        // 2. 更新本地变换
        this.updateLocalTransform();

        // 3. 更新全局变换
        this.updateGlobalTransform();

        // 4. 更新子骨骼
        this.updateChildren();

        // 5. 清除脏标志
        this._localDirty = false;
    }

    /**
     * 更新全局变换矩阵
     */
    private updateGlobalTransform(): void {
        // 如果有父骨骼，继承父骨骼的变换
        if (this._parent !== null) {
            this.global.copyFrom(this._parent.global);
            this.global.add(this.origin);
            this.global.add(this.offset);
            this.global.add(this.animationPose);
        } else {
            // 根骨骼直接使用本地变换
            this.global.copyFrom(this.origin);
            this.global.add(this.offset);
            this.global.add(this.animationPose);
        }
    }
}
```

### 3.2 骨架类实现

```typescript
/**
 * 骨架类 - 骨骼动画系统的核心
 *
 * 职责：
 * - 管理所有骨骼和插槽
 * - 协调动画播放
 * - 处理层级关系
 * - 提供对外接口
 *
 * 依赖：
 * - IAnimatable: 可动画接口
 * - BaseObject: 基础对象类
 */
export class Armature extends BaseObject implements IAnimatable {
    // 骨架属性
    public inheritAnimation: boolean;                 // 是否继承动画
    public userData: any;                             // 用户数据

    // 骨骼和插槽集合
    private readonly _bones: Array<Bone> = [];        // 骨骼数组
    private readonly _slots: Array<Slot> = [];        // 插槽数组
    public readonly _constraints: Array<Constraint> = []; // 约束数组

    // 更新控制
    private _slotsDirty: boolean;                     // 插槽脏标志
    private _zOrderDirty: boolean;                    // 层级脏标志
    public _zIndexDirty: boolean;                     // Z索引脏标志

    /**
     * 构建骨骼层级
     */
    public buildBones(): void {
        // 1. 创建所有骨骼实例
        for (const boneData of this._armatureData.sortedBones) {
            const bone = BaseObject.getObjectPool<Bone>().get(Bone);
            bone._init(boneData);
            this._bones.push(bone);
        }

        // 2. 建立骨骼父子关系
        for (const bone of this._bones) {
            if (bone.boneData.parent) {
                const parentBone = this.findBone(bone.boneData.parent);
                if (parentBone) {
                    bone._parent = parentBone;
                }
            }
        }
    }

    /**
     * 查找指定名称的骨骼
     */
    public findBone(name: string): Bone | null {
        for (const bone of this._bones) {
            if (bone.name === name) {
                return bone;
            }
        }
        return null;
    }

    /**
     * 更新骨架
     */
    public advanceTime(passedTime: number): void {
        // 1. 更新动画
        this._animation.advanceTime(passedTime);

        // 2. 更新骨骼
        this.updateBones();

        // 3. 更新插槽
        this.updateSlots();

        // 4. 更新约束
        this.updateConstraints();
    }

    /**
     * 更新所有骨骼
     */
    private updateBones(): void {
        // 按照层级顺序更新骨骼
        for (const bone of this._bones) {
            bone.update(false);
        }
    }
}
```

### 3.3 骨骼数据类实现

```typescript
/**
 * 骨架数据类 - 定义骨架的静态数据
 *
 * 职责：
 * - 存储骨架的配置信息
 * - 管理骨骼和动画数据
 * - 提供数据查询接口
 */
export class ArmatureData extends BaseObject {
    // 基本属性
    public type: ArmatureType;                        // 骨架类型
    public frameRate: number;                          // 帧率
    public name: string;                               // 骨架名称

    // 数据结构
    public readonly animationNames: Array<string> = []; // 动画名称列表
    public readonly sortedBones: Array<BoneData> = [];  // 排序的骨骼数据
    public readonly sortedSlots: Array<SlotData> = [];  // 排序的插槽数据

    /**
     * 添加骨骼数据
     */
    public addBoneData(boneData: BoneData): void {
        // 1. 添加到骨骼列表
        this._boneDataMap[boneData.name] = boneData;

        // 2. 重新排序骨骼
        this.sortBones();
    }

    /**
     * 骨骼排序算法
     */
    private sortBones(): void {
        this.sortedBones.length = 0;

        // 1. 找到所有根骨骼
        const rootBones: BoneData[] = [];
        for (const boneName in this._boneDataMap) {
            const boneData = this._boneDataMap[boneName];
            if (!boneData.parent || !this._boneDataMap[boneData.parent]) {
                rootBones.push(boneData);
            }
        }

        // 2. 深度优先遍历构建层级
        for (const rootBone of rootBones) {
            this.traverseBone(rootBone);
        }
    }

    /**
     * 遍历骨骼树
     */
    private traverseBone(boneData: BoneData): void {
        // 1. 添加当前骨骼
        this.sortedBones.push(boneData);

        // 2. 递归添加子骨骼
        for (const childBoneName in this._boneDataMap) {
            const childBone = this._boneDataMap[childBoneName];
            if (childBone.parent === boneData.name) {
                this.traverseBone(childBone);
            }
        }
    }
}
```

## 4. 关键算法

### 4.1 骨骼变换计算

```typescript
/**
 * 骨骼变换计算流程
 */
class BoneTransformCalculator {
    /**
     * 计算骨骼的全局变换
     */
    static calculateGlobalTransform(bone: Bone): Transform {
        // 1. 获取基础变换
        const transform = new Transform();
        transform.copyFrom(bone.origin);

        // 2. 添加偏移变换
        if (bone.offsetMode === OffsetMode.Additive) {
            transform.add(bone.offset);
        } else {
            transform.copyFrom(bone.offset);
        }

        // 3. 添加动画姿势
        transform.add(bone.animationPose);

        // 4. 应用父骨骼变换
        if (bone.parent) {
            const parentGlobal = this.calculateGlobalTransform(bone.parent);
            return Transform.add(parentGlobal, transform);
        }

        return transform;
    }
}
```

### 4.2 骨骼层级更新

```typescript
/**
 * 骨骼层级更新策略
 */
class BoneHierarchyUpdater {
    /**
     * 按照层级顺序更新骨骼
     */
    static updateBonesInOrder(bones: Bone[]): void {
        // 1. 拓扑排序确保父骨骼先更新
        const sortedBones = this.topologicalSort(bones);

        // 2. 依次更新每个骨骼
        for (const bone of sortedBones) {
            bone.update(false);
        }
    }

    /**
     * 拓扑排序算法
     */
    private static topologicalSort(bones: Bone[]): Bone[] {
        const result: Bone[] = [];
        const visited = new Set<Bone>();

        // 深度优先搜索
        function dfs(bone: Bone): void {
            if (visited.has(bone)) {
                return;
            }

            visited.add(bone);

            // 先访问父骨骼
            if (bone.parent) {
                dfs(bone.parent);
            }

            result.push(bone);
        }

        // 对每个骨骼执行DFS
        for (const bone of bones) {
            dfs(bone);
        }

        return result;
    }
}
```

## 5. 性能优化

### 5.1 脏检查机制

```typescript
/**
 * 脏检查优化策略
 */
class DirtyCheckOptimizer {
    /**
     * 优化骨骼更新
     */
    static optimizeBoneUpdate(bone: Bone): boolean {
        // 1. 检查本地变换是否需要更新
        if (!bone._localDirty) {
            return false;
        }

        // 2. 标记全局变换为脏
        bone._transformDirty = true;

        // 3. 标记所有子骨骼的全局变换为脏
        this.markChildrenDirty(bone);

        return true;
    }

    /**
     * 标记子骨骼为脏
     */
    private static markChildrenDirty(bone: Bone): void {
        for (const child of bone._children) {
            child._childrenTransformDirty = true;
            this.markChildrenDirty(child);
        }
    }
}
```

### 5.2 缓存优化

```typescript
/**
 * 缓存优化策略
 */
class CacheOptimizer {
    /**
     * 缓存帧索引优化动画查找
     */
    static optimizeAnimationLookup(bone: Bone, frameIndex: number): void {
        // 1. 检查缓存是否有效
        if (bone._cachedFrameIndex === frameIndex) {
            return; // 使用缓存结果
        }

        // 2. 更新缓存
        bone._cachedFrameIndex = frameIndex;

        // 3. 预计算下一帧索引
        bone._cachedFrameIndices = this.precalculateFrameIndices(bone, frameIndex);
    }

    /**
     * 预计算帧索引
     */
    private static precalculateFrameIndices(bone: Bone, currentFrame: number): number[] {
        const indices: number[] = [];

        // 预计算前后几帧的索引
        for (let i = -2; i <= 2; i++) {
            const frameIndex = currentFrame + i;
            if (frameIndex >= 0) {
                indices.push(frameIndex);
            }
        }

        return indices;
    }
}
```

## 8. 最佳实践

### 8.1 骨骼设计原则

1. **合理的骨骼层级** - 避免过深的层级嵌套
2. **适当的骨骼数量** - 平衡动画效果和性能
3. **正确的骨骼命名** - 使用清晰的命名规范
4. **优化骨骼更新** - 合理使用脏检查机制

### 8.2 性能优化建议

1. **减少骨骼数量** - 只在必要的地方添加骨骼
2. **使用对象池** - 重用骨骼对象减少内存分配
3. **批量更新** - 避免频繁的单个骨骼更新
4. **缓存计算结果** - 缓存频繁访问的数据

### 8.3 调试技巧

1. **骨骼可视化** - 开发时显示骨骼结构
2. **变换调试** - 显示骨骼的变换信息
3. **性能监控** - 监控骨骼更新的性能消耗
4. **层级检查** - 验证骨骼层级关系的正确性
