# 时间轴系统源码分析

## 1. 功能概述

时间轴系统是 DragonBones 动画制作的核心，负责宏观调节各元件动画的关键帧、播放速度、动画补间等细节。它提供了完整的动画时间控制机制，支持复杂的动画混合和状态管理。

### 核心特性
- **关键帧管理** - 支持多种关键帧类型和插值方式
- **播放控制** - 提供播放、暂停、跳转等控制功能
- **动画混合** - 支持多个动画状态的混合播放
- **时间缩放** - 可调节动画播放速度
- **循环模式** - 支持多种循环播放模式

## 3. 核心实现

### 3.1 动画控制器实现

```typescript
/**
 * 动画控制器 - 管理所有动画的播放和状态
 *
 * 职责：
 * - 管理多个动画状态
 * - 协调动画播放时序
 * - 处理动画混合
 * - 提供播放控制接口
 *
 * 依赖：
 * - AnimationState: 动画状态管理
 * - WorldClock: 时间管理
 * - AnimationData: 动画数据
 */
export class Animation extends BaseObject {
    // 时间控制
    public timeScale: number;                         // 全局时间缩放
    private _inheritTimeScale: number;                // 继承时间缩放

    // 动画状态管理
    private readonly _animationStates: Array<AnimationState> = []; // 动画状态数组
    private readonly _animations: Map<AnimationData> = {};        // 动画数据映射
    private _lastAnimationState: AnimationState | null;           // 最后播放的动画状态

    /**
     * 播放指定动画
     */
    public gotoAndPlay(name: string, time: number = 0): AnimationState | null {
        // 1. 获取动画数据
        const animationData = this._animations[name];
        if (!animationData) {
            return null;
        }

        // 2. 停止当前动画
        if (this._lastAnimationState) {
            this._lastAnimationState.stop();
        }

        // 3. 创建或获取动画状态
        let animationState = this.findAnimationState(name);
        if (!animationState) {
            animationState = this.createAnimationState(animationData);
            this._animationStates.push(animationState);
        }

        // 4. 配置动画状态
        animationState.currentTime = time;
        animationState.play();

        // 5. 更新最后播放的动画状态
        this._lastAnimationState = animationState;

        return animationState;
    }

    /**
     * 创建动画状态
     */
    private createAnimationState(animationData: AnimationData): AnimationState {
        const animationState = BaseObject.getObjectPool<AnimationState>().get(AnimationState);
        animationState._init(this._armature, animationData);
        return animationState;
    }

    /**
     * 查找动画状态
     */
    private findAnimationState(name: string): AnimationState | null {
        for (const animationState of this._animationStates) {
            if (animationState.name === name) {
                return animationState;
            }
        }
        return null;
    }

    /**
     * 更新时间轴
     */
    public advanceTime(passedTime: number): void {
        // 1. 应用时间缩放
        const scaledTime = passedTime * this.timeScale * this._inheritTimeScale;

        // 2. 更新所有动画状态
        for (const animationState of this._animationStates) {
            if (animationState.isPlaying) {
                animationState.advanceTime(scaledTime);
            }
        }

        // 3. 清理已完成的动画状态
        this.cleanupCompletedAnimations();
    }

    /**
     * 清理已完成的动画
     */
    private cleanupCompletedAnimations(): void {
        for (let i = this._animationStates.length - 1; i >= 0; i--) {
            const animationState = this._animationStates[i];
            if (animationState.isComplete) {
                animationState.returnToPool();
                this._animationStates.splice(i, 1);
            }
        }
    }
}
```

### 3.2 动画状态实现

```typescript
/**
 * 动画状态 - 管理单个动画的播放状态
 *
 * 职责：
 * - 管理动画播放进度
 * - 处理时间轴更新
 * - 执行动画混合
 * - 控制动画属性
 */
export class AnimationState extends BaseObject {
    // 播放控制属性
    public actionEnabled: boolean;                    // 是否启用动作
    public additive: boolean;                         // 是否叠加模式
    public displayControl: boolean;                   // 是否控制显示
    public resetToPose: boolean;                      // 是否重置到初始姿势

    // 时间属性
    public playTimes: number;                         // 播放次数
    public timeScale: number;                         // 时间缩放
    public blendType: AnimationBlendType;             // 混合类型

    // 内部状态
    private _currentTime: number;                     // 当前时间
    private _totalTime: number;                       // 总时间
    private _isPlaying: boolean;                      // 是否正在播放
    private _isComplete: boolean;                     // 是否完成
    private _progress: number;                        // 播放进度

    // 时间轴状态
    private readonly _timelineStates: Array<TimelineState> = []; // 时间轴状态数组

    /**
     * 获取当前时间
     */
    public get currentTime(): number {
        return this._currentTime;
    }

    public set currentTime(value: number) {
        this._currentTime = value;
        this._progress = this._currentTime / this._totalTime;
    }

    /**
     * 获取播放进度
     */
    public get progress(): number {
        return this._progress;
    }

    /**
     * 是否正在播放
     */
    public get isPlaying(): boolean {
        return this._isPlaying;
    }

    /**
     * 是否播放完成
     */
    public get isComplete(): boolean {
        return this._isComplete;
    }

    /**
     * 开始播放
     */
    public play(): void {
        this._isPlaying = true;
        this._isComplete = false;
    }

    /**
     * 停止播放
     */
    public stop(): void {
        this._isPlaying = false;
    }

    /**
     * 更新时间
     */
    public advanceTime(passedTime: number): void {
        if (!this._isPlaying) {
            return;
        }

        // 1. 应用时间缩放
        const scaledTime = passedTime * this.timeScale;

        // 2. 更新当前时间
        this._currentTime += scaledTime;

        // 3. 检查是否完成
        if (this._currentTime >= this._totalTime) {
            this.handleAnimationComplete();
        }

        // 4. 更新时间轴状态
        this.updateTimelineStates(scaledTime);

        // 5. 更新进度
        this._progress = this._currentTime / this._totalTime;
    }

    /**
     * 处理动画完成
     */
    private handleAnimationComplete(): void {
        if (this.playTimes > 0) {
            // 有限次播放
            this.playTimes--;
            if (this.playTimes <= 0) {
                this._isComplete = true;
                this._isPlaying = false;
            } else {
                // 重新开始
                this._currentTime = 0;
            }
        } else {
            // 无限循环
            this._currentTime = this._currentTime % this._totalTime;
        }
    }

    /**
     * 更新时间轴状态
     */
    private updateTimelineStates(passedTime: number): void {
        for (const timelineState of this._timelineStates) {
            timelineState.update(passedTime);
        }
    }
}
```

### 3.3 时间轴状态实现

```typescript
/**
 * 时间轴状态 - 管理单个时间轴的播放
 *
 * 职责：
 * - 管理关键帧数据
 * - 执行插值计算
 * - 更新时间轴属性
 */
export class TimelineState extends BaseObject {
    // 时间轴属性
    private _currentTime: number;                     // 当前时间
    private _frameCount: number;                      // 帧数量
    private _frames: FrameData[];                     // 帧数据数组

    // 插值相关
    private _tweenProgress: number;                   // 补间进度
    private _isTweenEnabled: boolean;                 // 是否启用补间

    /**
     * 更新时间轴
     */
    public update(passedTime: number): void {
        // 1. 更新当前时间
        this._currentTime += passedTime;

        // 2. 查找当前帧
        const currentFrame = this.findCurrentFrame(this._currentTime);

        // 3. 计算补间
        if (this._isTweenEnabled && currentFrame.nextFrame) {
            this.calculateTween(currentFrame, currentFrame.nextFrame);
        } else {
            this.applyFrame(currentFrame);
        }
    }

    /**
     * 查找当前帧
     */
    private findCurrentFrame(time: number): FrameData {
        // 二分查找优化
        let left = 0;
        let right = this._frameCount - 1;

        while (left <= right) {
            const mid = Math.floor((left + right) / 2);
            const frame = this._frames[mid];

            if (time < frame.time) {
                right = mid - 1;
            } else if (time > frame.time) {
                left = mid + 1;
            } else {
                return frame;
            }
        }

        // 返回最接近的帧
        return this._frames[Math.max(0, right)];
    }

    /**
     * 计算补间
     */
    private calculateTween(currentFrame: FrameData, nextFrame: FrameData): void {
        // 1. 计算补间进度
        const frameDuration = nextFrame.time - currentFrame.time;
        const elapsedTime = this._currentTime - currentFrame.time;
        this._tweenProgress = elapsedTime / frameDuration;

        // 2. 应用缓动函数
        this._tweenProgress = this.applyEasing(this._tweenProgress, currentFrame.easing);

        // 3. 插值计算
        const interpolatedValue = this.interpolateValues(
            currentFrame.value,
            nextFrame.value,
            this._tweenProgress
        );

        // 4. 应用插值结果
        this.applyValue(interpolatedValue);
    }

    /**
     * 应用缓动函数
     */
    private applyEasing(progress: number, easing: number): number {
        if (easing === 0) {
            return progress; // 线性
        } else if (easing > 0) {
            // 二次缓动
            const t = progress;
            return t * t * (3 - 2 * t);
        } else {
            // 弹性缓动
            const t = progress;
            return t * t * t * (t * (t * 6 - 15) + 10);
        }
    }

    /**
     * 值插值计算
     */
    private interpolateValues(startValue: any, endValue: any, progress: number): any {
        if (typeof startValue === 'number' && typeof endValue === 'number') {
            // 数值插值
            return startValue + (endValue - startValue) * progress;
        } else if (startValue instanceof Transform && endValue instanceof Transform) {
            // 变换插值
            const result = new Transform();
            result.x = startValue.x + (endValue.x - startValue.x) * progress;
            result.y = startValue.y + (endValue.y - startValue.y) * progress;
            result.rotation = startValue.rotation + (endValue.rotation - startValue.rotation) * progress;
            result.scaleX = startValue.scaleX + (endValue.scaleX - startValue.scaleX) * progress;
            result.scaleY = startValue.scaleY + (endValue.scaleY - startValue.scaleY) * progress;
            return result;
        }

        // 其他类型直接返回起始值
        return startValue;
    }
}
```

## 4. 关键算法

### 4.1 关键帧插值算法

```typescript
/**
 * 关键帧插值算法
 */
class KeyframeInterpolation {
    /**
     * 贝塞尔曲线插值
     */
    static bezierInterpolation(t: number, p1: number, p2: number, p3: number, p4: number): number {
        const u = 1 - t;
        const tt = t * t;
        const uu = u * u;
        const uuu = uu * u;
        const ttt = tt * t;

        return uuu * p1 + 3 * uu * t * p2 + 3 * u * tt * p3 + ttt * p4;
    }

     /**
      * 样条插值
      */
    static splineInterpolation(frames: FrameData[], time: number): number {
        // 查找相邻帧
        const frameIndex = this.findFrameIndex(frames, time);
        if (frameIndex < 0) return frames[0].value;
        if (frameIndex >= frames.length - 1) return frames[frames.length - 1].value;

        const currentFrame = frames[frameIndex];
        const nextFrame = frames[frameIndex + 1];

        // 计算插值参数
        const t = (time - currentFrame.time) / (nextFrame.time - currentFrame.time);

        // 应用样条插值
        return this.catmullRomInterpolation(
            frameIndex > 0 ? frames[frameIndex - 1].value : currentFrame.value,
            currentFrame.value,
            nextFrame.value,
            frameIndex < frames.length - 2 ? frames[frameIndex + 2].value : nextFrame.value,
            t
        );
    }

    /**
     * Catmull-Rom 样条插值
     */
    private static catmullRomInterpolation(p0: number, p1: number, p2: number, p3: number, t: number): number {
        const t2 = t * t;
        const t3 = t2 * t;

        return 0.5 * (
            (2 * p1) +
            (-p0 + p2) * t +
            (2 * p0 - 5 * p1 + 4 * p2 - p3) * t2 +
            (-p0 + 3 * p1 - 3 * p2 + p3) * t3
        );
    }
}
```

### 4.2 动画混合算法

```typescript
/**
 * 动画混合算法
 */
class AnimationBlending {
    /**
     * 线性混合
     */
    static linearBlend(animations: AnimationState[], weights: number[]): Transform {
        const result = new Transform();
        let totalWeight = 0;

        for (let i = 0; i < animations.length; i++) {
            const weight = weights[i];
            if (weight > 0) {
                const animation = animations[i];
                const transform = animation.getCurrentTransform();

                result.x += transform.x * weight;
                result.y += transform.y * weight;
                result.rotation += transform.rotation * weight;
                result.scaleX += transform.scaleX * weight;
                result.scaleY += transform.scaleY * weight;

                totalWeight += weight;
            }
        }

        // 归一化
        if (totalWeight > 0) {
            result.x /= totalWeight;
            result.y /= totalWeight;
            result.rotation /= totalWeight;
            result.scaleX /= totalWeight;
            result.scaleY /= totalWeight;
        }

        return result;
    }

    /**
     * 加法混合
     */
    static additiveBlend(baseAnimation: AnimationState, additiveAnimations: AnimationState[], weights: number[]): Transform {
        // 1. 获取基础变换
        const result = baseAnimation.getCurrentTransform().clone();

        // 2. 添加加法动画
        for (let i = 0; i < additiveAnimations.length; i++) {
            const weight = weights[i];
            if (weight > 0) {
                const additiveAnimation = additiveAnimations[i];
                const additiveTransform = additiveAnimation.getCurrentTransform();

                // 只添加相对变换
                result.x += additiveTransform.x * weight;
                result.y += additiveTransform.y * weight;
                result.rotation += additiveTransform.rotation * weight;
            }
        }

        return result;
    }
}
```

## 8. 最佳实践

### 8.1 时间轴设计原则

1. **合理的关键帧密度** - 避免过多或过少的关键帧
2. **适当的插值方式** - 根据动画特性选择合适的插值算法
3. **优化时间轴结构** - 减少不必要的时间轴数量
4. **合理使用混合** - 避免过度复杂的动画混合

### 8.2 性能优化建议

1. **减少时间轴数量** - 合并相关的时间轴
2. **使用对象池** - 重用时间轴状态对象
3. **批量更新** - 避免频繁的单个时间轴更新
4. **缓存计算结果** - 缓存频繁访问的时间轴数据

### 8.3 调试技巧

1. **时间轴可视化** - 显示时间轴的关键帧和播放进度
2. **性能监控** - 监控时间轴更新的性能消耗
3. **事件调试** - 显示时间轴事件触发情况
4. **混合调试** - 显示动画混合的效果和权重
