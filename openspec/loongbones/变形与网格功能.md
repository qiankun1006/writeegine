# 变形与网格功能源码分析

## 1. 功能概述

变形与网格功能是 DragonBones 的高级特性，主要包括网格和自由变形系统、网格变形器、矢量形状系统三个核心模块。这些功能提供了类似 Live2D 的变形效果，支持复杂的网格动画和形状变换。

### 核心特性
- **网格变形** - 支持任意四边形网格的自由变形
- **骨骼绑定** - 网格顶点可以绑定到骨骼进行动画控制
- **实时变形** - 支持运行时动态网格变形
- **矢量形状** - 支持矢量图形的变形动画
- **性能优化** - 采用高效的网格计算和缓存机制

## 3. 核心实现

### 3.1 表面骨骼实现

```typescript
/**
 * 表面骨骼 - 实现网格变形的核心类
 *
 * 职责：
 * - 管理网格顶点数据
 * - 计算顶点变换矩阵
 * - 处理骨骼绑定和权重
 * - 执行实时网格变形
 *
 * 依赖：
 * - Bone: 基础骨骼类
 * - SurfaceData: 表面数据
 * - Matrix: 矩阵计算
 */
export class Surface extends Bone {
    // 网格数据
    public readonly _vertices: Array<number> = [];           // 顶点数组
    public readonly _deformVertices: Array<number> = [];     // 变形顶点数组

    // 缓存数据
    private readonly _hullCache: Array<number> = [];         // 边界缓存
    private readonly _matrixCahce: Array<number> = [];       // 矩阵缓存

    // 网格参数
    private _dX: number;                                     // X方向分段大小
    private _dY: number;                                     // Y方向分段大小
    private _k: number;                                      // 斜率参数
    private _kX: number;                                     // X斜率
    private _kY: number;                                     // Y斜率

    // 骨骼绑定
    public _bone: Bone | null;                               // 绑定的骨骼

    /**
     * 初始化表面骨骼
     */
    public init(surfaceData: SurfaceData, armatureValue: Armature): void {
        super.init(surfaceData, armatureValue);

        // 1. 计算网格参数
        const segmentX = surfaceData.segmentX;
        const segmentY = surfaceData.segmentY;
        const lA = 200.0; // 网格范围
        const lB = 1000.0; // 扩展范围

        this._dX = lA * 2.0 / segmentX;  // X方向分段大小
        this._dY = lA * 2.0 / segmentY;  // Y方向分段大小
        this._k = -this._dY / this._dX;  // 中心区域斜率
        this._kX = -this._dY / (lB - lA); // X边界斜率
        this._kY = -(lB - lA) / this._dX; // Y边界斜率

        // 2. 初始化顶点数组
        const vertexCount = this._armature.armatureData.parent.intArray[
            surfaceData.geometry.offset + BinaryOffset.GeometryVertexCount
        ];

        this._vertices.length = vertexCount * 2;
        this._deformVertices.length = vertexCount * 2;

        // 3. 初始化缓存数组
        this._matrixCahce.length = (segmentX * segmentY + segmentX * 2 + segmentY * 2) * 2 * 7;
        this._hullCache.length = 12;

        // 4. 重置变形顶点
        for (let i = 0; i < vertexCount * 2; ++i) {
            this._deformVertices[i] = 0.0;
        }

        // 5. 绑定骨骼
        if (this._parent !== null) {
            if (this._parent.boneData.type === BoneType.Bone) {
                this._bone = this._parent;
            } else {
                this._bone = (this._parent as Surface)._bone;
            }
        }
    }

    /**
     * 更新顶点数据
     */
    private _updateVertices(): void {
        const data = this._armature.armatureData.parent;
        const geometry = (this._boneData as SurfaceData).geometry;
        const intArray = data.intArray;
        const floatArray = data.floatArray;
        const vertexCount = intArray[geometry.offset + BinaryOffset.GeometryVertexCount];
        const verticesOffset = intArray[geometry.offset + BinaryOffset.GeometryFloatOffset];

        const vertices = this._vertices;
        const animationVertices = this._deformVertices;

        if (this._parent !== null) {
            if (this._parent._boneData.type === BoneType.Surface) {
                // 父骨骼也是表面骨骼
                for (let i = 0, l = vertexCount; i < l; ++i) {
                    const iD = i * 2;
                    const x = floatArray[verticesOffset + iD] + animationVertices[iD];
                    const y = floatArray[verticesOffset + iD + 1] + animationVertices[iD + 1];
                    const matrix = (this._parent as Surface)._getGlobalTransformMatrix(x, y);

                    vertices[iD] = matrix.a * x + matrix.c * y + matrix.tx;
                    vertices[iD + 1] = matrix.b * x + matrix.d * y + matrix.ty;
                }
            } else {
                // 父骨骼是普通骨骼
                const parentMatrix = this._parent.globalTransformMatrix;
                for (let i = 0, l = vertexCount; i < l; ++i) {
                    const iD = i * 2;
                    const x = floatArray[verticesOffset + iD] + animationVertices[iD];
                    const y = floatArray[verticesOffset + iD + 1] + animationVertices[iD + 1];

                    vertices[iD] = parentMatrix.a * x + parentMatrix.c * y + parentMatrix.tx;
                    vertices[iD + 1] = parentMatrix.b * x + parentMatrix.d * y + parentMatrix.ty;
                }
            }
        } else {
            // 没有父骨骼
            for (let i = 0, l = vertexCount; i < l; ++i) {
                const iD = i * 2;
                vertices[iD] = floatArray[verticesOffset + iD] + animationVertices[iD];
                vertices[iD + 1] = floatArray[verticesOffset + iD + 1] + animationVertices[iD + 1];
            }
        }
    }

    /**
     * 获取全局变换矩阵
     */
    public _getGlobalTransformMatrix(x: number, y: number): Matrix {
        const lA = 200.0;
        const lB = 1000.0;

        // 检查是否在有效范围内
        if (x < -lB || lB < x || y < -lB || lB < y) {
            return this.globalTransformMatrix;
        }

        let isDown = false;
        const surfaceData = this._boneData as SurfaceData;
        const segmentX = surfaceData.segmentX;
        const segmentY = surfaceData.segmentY;
        const segmentXD = surfaceData.segmentX * 2;

        const dX = this._dX;
        const dY = this._dY;
        const indexX = Math.floor((x + lA) / dX);
        const indexY = Math.floor((y + lA) / dY);

        let matrixIndex = 0;
        let pX = indexX * dX - lA;
        let pY = indexY * dY - lA;

        const matrices = this._matrixCahce;
        const helpMatrix = Surface._helpMatrix;

        // 根据位置选择不同的变换区域
        if (x < -lA) {
            // 左边界区域
            this.handleLeftBoundary(x, y, pX, pY, indexY, matrices, helpMatrix, segmentX, segmentY, segmentXD);
        } else if (x >= lA) {
            // 右边界区域
            this.handleRightBoundary(x, y, pX, pY, indexY, matrices, helpMatrix, segmentX, segmentY, segmentXD);
        } else if (y < -lA) {
            // 上边界区域
            this.handleTopBoundary(x, y, pX, pY, indexX, matrices, helpMatrix, segmentX, segmentY);
        } else if (y >= lA) {
            // 下边界区域
            this.handleBottomBoundary(x, y, pX, pY, indexX, matrices, helpMatrix, segmentX, segmentY, segmentXD);
        } else {
            // 中心区域
            this.handleCenterRegion(x, y, pX, pY, indexX, indexY, matrices, helpMatrix, segmentX, segmentXD);
        }

        return helpMatrix;
    }

    /**
     * 处理左边界区域
     */
    private handleLeftBoundary(
        x: number, y: number, pX: number, pY: number, indexY: number,
        matrices: number[], helpMatrix: Matrix, segmentX: number, segmentY: number, segmentXD: number
    ): void {
        if (y < -200 || y >= 200) {
            return; // 超出范围
        }

        const isDown = y > this._kX * (x + 200) + pY;
        const matrixIndex = ((segmentX * segmentY + segmentX + segmentY + segmentY + indexY) * 2 + (isDown ? 1 : 0)) * 7;

        if (matrices[matrixIndex] > 0.0) {
            helpMatrix.copyFromArray(matrices, matrixIndex + 1);
        } else {
            this.calculateLeftBoundaryMatrix(isDown, pY, indexY, matrices, helpMatrix, matrixIndex, segmentXD);
        }
    }

    /**
     * 计算左边界变换矩阵
     */
    private calculateLeftBoundaryMatrix(
        isDown: boolean, pY: number, indexY: number,
        matrices: number[], helpMatrix: Matrix, matrixIndex: number, segmentXD: number
    ): void {
        const vertexIndex = indexY * (segmentXD + 2);
        const ddX = this._hullCache[4];
        const ddY = this._hullCache[5];
        const sX = this._hullCache[2] - (indexY - indexY) * ddX;
        const sY = this._hullCache[3] - (indexY - indexY) * ddY;
        const vertices = this._vertices;

        if (isDown) {
            this._getAffineTransform(
                -200, pY + this._dY, 1000 - 200, this._dY,
                vertices[vertexIndex + segmentXD + 2],
                vertices[vertexIndex + segmentXD + 3],
                sX + ddX, sY + ddY,
                vertices[vertexIndex],
                vertices[vertexIndex + 1],
                Surface._helpTransform, helpMatrix, true
            );
        } else {
            this._getAffineTransform(
                -1000, pY, 1000 - 200, this._dY,
                sX, sY,
                vertices[vertexIndex],
                vertices[vertexIndex + 1],
                sX + ddX, sY + ddY,
                Surface._helpTransform, helpMatrix, false
            );
        }

        this.cacheMatrix(matrices, matrixIndex, helpMatrix);
    }

    /**
     * 缓存变换矩阵
     */
    private cacheMatrix(matrices: number[], matrixIndex: number, matrix: Matrix): void {
        matrices[matrixIndex] = 1.0;
        matrices[matrixIndex + 1] = matrix.a;
        matrices[matrixIndex + 2] = matrix.b;
        matrices[matrixIndex + 3] = matrix.c;
        matrices[matrixIndex + 4] = matrix.d;
        matrices[matrixIndex + 5] = matrix.tx;
        matrices[matrixIndex + 6] = matrix.ty;
    }
}
```

### 3.2 网格变形器实现

```typescript
/**
 * 网格变形器 - 实现高级网格变形功能
 *
 * 职责：
 * - 计算顶点权重
 * - 执行网格变形
 * - 优化网格质量
 * - 提供变形动画支持
 */
class MeshDeformer {
    private _weightCache = new Map<string, WeightData>();
    private _deformCache = new Map<string, MeshData>();

    /**
     * 计算顶点权重
     */
    public calculateWeights(vertices: number[], bones: Bone[]): WeightData {
        const cacheKey = this.generateWeightCacheKey(vertices, bones);

        if (this._weightCache.has(cacheKey)) {
            return this._weightCache.get(cacheKey)!;
        }

        const weightData = new WeightData();
        const vertexCount = vertices.length / 2;

        // 为每个顶点计算权重
        for (let i = 0; i < vertexCount; i++) {
            const x = vertices[i * 2];
            const y = vertices[i * 2 + 1];

            // 计算到每个骨骼的距离
            const distances: number[] = [];
            for (const bone of bones) {
                const distance = this.calculateDistanceToBone(x, y, bone);
                distances.push(distance);
            }

            // 根据距离计算权重
            const weights = this.calculateWeightsFromDistances(distances);

            weightData.vertexWeights.push(weights);
            weightData.boneIndices.push(this.getInfluentialBones(weights));
        }

        this._weightCache.set(cacheKey, weightData);
        return weightData;
    }

    /**
     * 计算到骨骼的距离
     */
    private calculateDistanceToBone(x: number, y: number, bone: Bone): number {
        const boneX = bone.global.x;
        const boneY = bone.global.y;
        const dx = x - boneX;
        const dy = y - boneY;

        return Math.sqrt(dx * dx + dy * dy);
    }

    /**
     * 根据距离计算权重
     */
    private calculateWeightsFromDistances(distances: number[]): number[] {
        const weights: number[] = [];
        let totalInverseDistance = 0;

        // 计算距离的倒数
        for (const distance of distances) {
            const inverseDistance = 1 / (distance + 1); // 加1避免除零
            weights.push(inverseDistance);
            totalInverseDistance += inverseDistance;
        }

        // 归一化权重
        for (let i = 0; i < weights.length; i++) {
            weights[i] /= totalInverseDistance;
        }

        return weights;
    }

    /**
     * 获取有影响力的骨骼
     */
    private getInfluentialBones(weights: number[]): number[] {
        const influentialBones: number[] = [];
        const threshold = 0.1; // 权重阈值

        for (let i = 0; i < weights.length; i++) {
            if (weights[i] > threshold) {
                influentialBones.push(i);
            }
        }

        return influentialBones;
    }

    /**
     * 执行网格变形
     */
    public deformMesh(meshData: MeshData, bones: Bone[]): MeshData {
        const deformedMesh = new MeshData();
        deformedMesh.vertices = [...meshData.vertices];
        deformedMesh.triangles = [...meshData.triangles];
        deformedMesh.uvs = [...meshData.uvs];

        // 计算权重
        const weightData = this.calculateWeights(meshData.vertices, bones);

        // 变形顶点
        const vertexCount = meshData.vertices.length / 2;
        for (let i = 0; i < vertexCount; i++) {
            const x = meshData.vertices[i * 2];
            const y = meshData.vertices[i * 2 + 1];

            // 计算变形后的位置
            const deformedPosition = this.calculateDeformedPosition(
                x, y, bones, weightData.vertexWeights[i]
            );

            deformedMesh.vertices[i * 2] = deformedPosition.x;
            deformedMesh.vertices[i * 2 + 1] = deformedPosition.y;
        }

        return deformedMesh;
    }

    /**
     * 计算变形后的位置
     */
    private calculateDeformedPosition(
        x: number, y: number, bones: Bone[], weights: number[]
    ): { x: number; y: number } {
        let finalX = 0;
        let finalY = 0;

        // 根据权重混合骨骼变换
        for (let i = 0; i < bones.length && i < weights.length; i++) {
            const weight = weights[i];
            if (weight > 0) {
                const bone = bones[i];
                const transformedPos = bone.globalTransformMatrix.transformPoint(x, y);

                finalX += transformedPos.x * weight;
                finalY += transformedPos.y * weight;
            }
        }

        return { x: finalX, y: finalY };
    }

    /**
     * 网格优化
     */
    public optimizeMesh(meshData: MeshData): MeshData {
        const optimizedMesh = new MeshData();

        // 1. 移除重复顶点
        const uniqueVertices = this.removeDuplicateVertices(meshData.vertices);

        // 2. 优化三角形质量
        const optimizedTriangles = this.optimizeTriangleQuality(uniqueVertices, meshData.triangles);

        // 3. 网格平滑
        const smoothedVertices = this.laplacianSmooth(uniqueVertices, optimizedTriangles);

        optimizedMesh.vertices = smoothedVertices;
        optimizedMesh.triangles = optimizedTriangles;
        optimizedMesh.uvs = this.updateUVs(meshData.uvs, meshData.vertices, smoothedVertices);

        return optimizedMesh;
    }

    /**
     * Laplacian平滑
     */
    private laplacianSmooth(vertices: number[], triangles: number[]): number[] {
        const smoothedVertices = [...vertices];
        const iterations = 3;

        for (let iter = 0; iter < iterations; iter++) {
            for (let i = 0; i < vertices.length; i += 2) {
                const neighbors = this.findVertexNeighbors(i / 2, triangles);

                if (neighbors.length > 0) {
                    let avgX = 0;
                    let avgY = 0;

                    for (const neighbor of neighbors) {
                        avgX += vertices[neighbor * 2];
                        avgY += vertices[neighbor * 2 + 1];
                    }

                    smoothedVertices[i] = avgX / neighbors.length;
                    smoothedVertices[i + 1] = avgY / neighbors.length;
                }
            }

            // 更新顶点数据
            for (let i = 0; i < vertices.length; i++) {
                vertices[i] = smoothedVertices[i];
            }
        }

        return smoothedVertices;
    }

    /**
     * 查找顶点邻居
     */
    private findVertexNeighbors(vertexIndex: number, triangles: number[]): number[] {
        const neighbors = new Set<number>();

        for (let i = 0; i < triangles.length; i += 3) {
            const v1 = triangles[i];
            const v2 = triangles[i + 1];
            const v3 = triangles[i + 2];

            if (v1 === vertexIndex) {
                neighbors.add(v2);
                neighbors.add(v3);
            } else if (v2 === vertexIndex) {
                neighbors.add(v1);
                neighbors.add(v3);
            } else if (v3 === vertexIndex) {
                neighbors.add(v1);
                neighbors.add(v2);
            }
        }

        return Array.from(neighbors);
    }
}
```

```typescript
/**
 * 网格变形核心算法

class MeshDeformationAlgorithms {
    /**
     * 双线性插值
     */
    static bilinearInterpolation(
        x: number, y: number,
        p00: number, p10: number, p01: number, p11: number
    ): number {
        const x1 = Math.floor(x);
        const x2 = Math.ceil(x);
        const y1 = Math.floor(y);
        const y2 = Math.ceil(y);

        const wx = x - x1;
        const wy = y - y1;

        const i1 = p00 * (1 - wx) + p10 * wx;
        const i2 = p01 * (1 - wx) + p11 * wx;

        return i1 * (1 - wy) + i2 * wy;
    }

    /**
     * 重心坐标计算
     */
    static calculateBarycentricCoordinates(
        point: { x: number; y: number },
        triangle: { x: number; y: number }[]
    ): { u: number; v: number; w: number } {
        const v0 = { x: triangle[1].x - triangle[0].x, y: triangle[1].y - triangle[0].y };
        const v1 = { x: triangle[2].x - triangle[0].x, y: triangle[2].y - triangle[0].y };
        const v2 = { x: point.x - triangle[0].x, y: point.y - triangle[0].y };

        const dot00 = v0.x * v0.x + v0.y * v0.y;
        const dot01 = v0.x * v1.x + v0.y * v1.y;
        const dot02 = v0.x * v2.x + v0.y * v2.y;
        const dot11 = v1.x * v1.x + v1.y * v1.y;
        const dot12 = v1.x * v2.x + v1.y * v2.y;

        const invDenom = 1 / (dot00 * dot11 - dot01 * dot01);
        const u = (dot11 * dot02 - dot01 * dot12) * invDenom;
        const v = (dot00 * dot12 - dot01 * dot02) * invDenom;
        const w = 1 - u - v;

        return { u, v, w };
    }

    /**
     * 网格细分算法
     */
    static subdivideMesh(meshData: MeshData, iterations: number): MeshData {
        let subdividedMesh = meshData;

        for (let i = 0; i < iterations; i++) {
            subdividedMesh = this.catmullClarkSubdivision(subdividedMesh);
        }

        return subdividedMesh;
    }

    /**
     * Catmull-Clark细分
     */
    private static catmullClarkSubdivision(meshData: MeshData): MeshData {
        const newVertices: number[] = [];
        const newTriangles: number[] = [];

        // 1. 计算面点
        const facePoints = this.calculateFacePoints(meshData);

        // 2. 计算边点
        const edgePoints = this.calculateEdgePoints(meshData, facePoints);

        // 3. 计算新的顶点位置
        const updatedVertices = this.updateVertexPositions(meshData, facePoints, edgePoints);

        // 4. 构建新的网格
        return this.buildSubdividedMesh(updatedVertices, edgePoints, facePoints);
    }
}
```

```typescript
/**
 * 权重计算算法
 */
class WeightCalculationAlgorithms {
    /**
     * 线性混合权重
     */
    static linearBlendWeights(weights: number[], transforms: Matrix[]): Matrix {
        const resultMatrix = new Matrix();
        let totalWeight = 0;

        for (let i = 0; i < weights.length; i++) {
            const weight = weights[i];
            if (weight > 0) {
                const transform = transforms[i];

                resultMatrix.a += transform.a * weight;
                resultMatrix.b += transform.b * weight;
                resultMatrix.c += transform.c * weight;
                resultMatrix.d += transform.d * weight;
                resultMatrix.tx += transform.tx * weight;
                resultMatrix.ty += transform.ty * weight;

                totalWeight += weight;
            }
        }

        if (totalWeight > 0) {
            resultMatrix.a /= totalWeight;
            resultMatrix.b /= totalWeight;
            resultMatrix.c /= totalWeight;
            resultMatrix.d /= totalWeight;
            resultMatrix.tx /= totalWeight;
            resultMatrix.ty /= totalWeight;
        }

        return resultMatrix;
    }

    /**
     * 距离衰减权重
     */
    static distanceBasedWeights(
        vertex: { x: number; y: number },
        bones: { x: number; y: number; influenceRadius: number }[]
    ): number[] {
        const weights: number[] = [];
        let totalWeight = 0;

        for (const bone of bones) {
            const distance = Math.sqrt(
                (vertex.x - bone.x) ** 2 + (vertex.y - bone.y) ** 2
            );

            let weight = 0;
            if (distance <= bone.influenceRadius) {
                // 使用平滑衰减函数
                const t = distance / bone.influenceRadius;
                weight = (1 - t) ** 2 * (1 + 2 * t); // smoothstep函数
            }

            weights.push(weight);
            totalWeight += weight;
        }

        // 归一化
        if (totalWeight > 0) {
            for (let i = 0; i < weights.length; i++) {
                weights[i] /= totalWeight;
            }
        }

        return weights;
    }
}
```

### 5.2 批量处理优化

```typescript
/**
 * 批量网格处理优化
 */
class BatchMeshProcessor {
    private static _processingQueue: MeshJob[] = [];
    private static _isProcessing = false;

    /**
     * 添加处理任务
     */
    static addJob(job: MeshJob): void {
        this._processingQueue.push(job);

        if (!this._isProcessing) {
            this.processQueue();
        }
    }

    /**
     * 处理任务队列
     */
    private static async processQueue(): Promise<void> {
        this._isProcessing = true;

        while (this._processingQueue.length > 0) {
            const batch = this._processingQueue.splice(0, 10); // 每次处理10个任务

            await Promise.all(
                batch.map(job => this.processJob(job))
            );
        }

        this._isProcessing = false;
    }

    /**
     * 处理单个任务
     */
    private static async processJob(job: MeshJob): Promise<void> {
        try {
            const result = await this.deformMeshJob(job);
            job.onComplete(result);
        } catch (error) {
            job.onError(error);
        }
    }

    /**
     * 网格变形任务处理
     */
    private static async deformMeshJob(job: MeshJob): Promise<MeshData> {
        const { meshData, bones, options } = job;

        // 使用Web Worker进行并行处理
        if (typeof Worker !== 'undefined') {
            return await this.processInWorker(meshData, bones, options);
        } else {
            // 回退到主线程处理
            const deformer = new MeshDeformer();
            return deformer.deformMesh(meshData, bones);
        }
    }

    /**
     * 在Web Worker中处理
     */
    private static async processInWorker(
        meshData: MeshData, bones: Bone[], options: DeformOptions
    ): Promise<MeshData> {
        return new Promise((resolve, reject) => {
            const worker = new Worker('/mesh-deform-worker.js');

            worker.postMessage({ meshData, bones, options });

            worker.onmessage = (e) => {
                resolve(e.data);
                worker.terminate();
            };

            worker.onerror = (error) => {
                reject(error);
                worker.terminate();
            };
        });
    }
}
```

### 8.1 网格设计原则

1. **合理的网格密度** - 根据变形需求选择合适的网格密度
