# 骨骼与动力学功能源码分析

## 1. 功能概述

骨骼与动力学功能是 DragonBones 的高级动画系统，主要包括 IK 骨骼约束、骨骼权重和蒙皮动画、骨架嵌套系统、路径约束、物理约束、变换约束等多个子系统。这些功能提供了类似专业 3D 软件的骨骼动力学效果。

### 核心特性
- **IK反向动力学** - 支持单链和复杂链的IK求解
- **骨骼权重** - 精确的顶点权重绑定和蒙皮变形
- **骨架嵌套** - 支持多层骨架的嵌套和复用
- **路径约束** - 骨骼沿路径运动的约束系统
- **物理约束** - 重力、风力等物理效果模拟
- **变换约束** - 骨骼间的变换关系约束

### 3.2 IK约束实现

```typescript
/**
 * IK反向动力学约束 - 实现骨骼的IK求解
 *
 * 职责：
 * - 计算IK链的骨骼旋转
 * - 处理多骨骼IK求解
 * - 支持不同的求解算法
 *
 * 依赖：
 * - Constraint: 约束基类
 * - Bone: 骨骼类
 * - Transform: 变换计算
 */
export class IKConstraint extends Constraint {
    public static toString(): string {
        return "[class dragonBones.IKConstraint]";
    }

    // IK参数
    public _bendPositive: boolean;   // 弯曲方向
    public _weight: number;          // 约束权重

    /**
     * 初始化IK约束
     */
    public init(constraintData: ConstraintData, armature: Armature): void {
        if (this._constraintData) {
            return;
        }

        super.init(constraintData, armature);

        // 1. 设置约束数据
        this._constraintData = constraintData;
        this._armature = armature;

        // 2. 获取目标骨骼
        this._target = armature.findBone(constraintData.target.name);
        if (!this._target) {
            throw new Error(`Target bone not found: ${constraintData.target.name}`);
        }

        // 3. 获取根骨骼
        this._root = armature.findBone(constraintData.root.name);
        if (!this._root) {
            throw new Error(`Root bone not found: ${constraintData.root.name}`);
        }

        // 4. 获取影响的骨骼
        if (constraintData.bone) {
            this._bone = armature.findBone(constraintData.bone.name);
        }

        // 5. 设置IK参数
        const ikData = constraintData as IKConstraintData;
        this._bendPositive = ikData.bendPositive;
        this._weight = ikData.weight;
    }

    /**
     * 更新IK约束
     */
    public update(): void {
        // 1. 检查权重
        if (this._weight === 0) {
            return;
        }

        // 2. 根据骨骼数量选择求解方法
        if (this._bone) {
            // 两骨骼IK求解
            this._computeB();
        } else {
            // 单骨骼IK求解
            this._computeA();
        }
    }

    /**
     * 单骨骼IK求解
     */
    private _computeA(): void {
        const ikGlobal = this._target.global;
        const global = this._root.global;
        const globalTransformMatrix = this._root.globalTransformMatrix;

        // 1. 计算目标角度
        let radian = Math.atan2(ikGlobal.y - global.y, ikGlobal.x - global.x);

        // 2. 处理负缩放
        if (global.scaleX < 0.0) {
            radian += Math.PI;
        }

        // 3. 应用权重和角度归一化
        global.rotation += Transform.normalizeRadian(radian - global.rotation) * this._weight;

        // 4. 更新变换矩阵
        global.toMatrix(globalTransformMatrix);
    }

    /**
     * 两骨骼IK求解 (CCD算法)
     */
    private _computeB(): void {
        const boneLength = (this._bone as Bone)._boneData.length;
        const parent = this._root as Bone;
        const ikGlobal = this._target.global;
        const parentGlobal = parent.global;
        const global = (this._bone as Bone).global;
        const globalTransformMatrix = (this._bone as Bone).globalTransformMatrix;

        // 1. 计算骨骼长度
        const x = globalTransformMatrix.a * boneLength;
        const y = globalTransformMatrix.b * boneLength;
        const lLL = x * x + y * y;
        const lL = Math.sqrt(lLL);

        // 2. 计算父骨骼到当前骨骼的距离
        let dX = global.x - parentGlobal.x;
        let dY = global.y - parentGlobal.y;
        const lPP = dX * dX + dY * dY;
        const lP = Math.sqrt(lPP);

        // 3. 计算目标距离
        dX = ikGlobal.x - parentGlobal.x;
        dY = ikGlobal.y - parentGlobal.y;
        const lTT = dX * dX + dY * dY;
        const lT = Math.sqrt(lTT);

        // 4. 检查是否可达
        let radianA = 0.0;
        if (lL + lP <= lT || lT + lL <= lP || lT + lP <= lL) {
            // 不可达，直接指向目标
            radianA = Math.atan2(ikGlobal.y - parentGlobal.y, ikGlobal.x - parentGlobal.x);
            if (lL + lP <= lT) {
                // 完全伸展
            } else if (lP < lL) {
                // 反向弯曲
                radianA += Math.PI;
            }
        } else {
            // 使用余弦定理计算角度
            const h = (lPP - lLL + lTT) / (2.0 * lTT);
            const r = Math.sqrt(lPP - h * h * lTT) / lT;
            const hX = parentGlobal.x + (dX * h);
            const hY = parentGlobal.y + (dY * h);
            const rX = -dY * r;
            const rY = dX * r;

            // 选择弯曲方向
            let isPPR = false;
            const parentParent = parent.parent;
            if (parentParent) {
                const parentParentGlobal = parentParent.global;
                isPPR = (parentParentGlobal.x - global.x) * (global.y - hY) -
                       (parentParentGlobal.y - global.y) * (global.x - hX) > 0;
            }

            // 计算最终位置
            const currentRadian = Math.atan2(
                hY + (this._bendPositive === isPPR ? rY : -rY) - global.y,
                hX + (this._bendPositive === isPPR ? rX : -rX) - global.x
            );

            radianA = Transform.normalizeRadian(currentRadian - Math.atan2(dY, dX));
        }

        // 5. 更新骨骼旋转
        const currentRadian = Math.atan2(dY, dX);
        const parentRadian = parentGlobal.rotation;

        parentGlobal.rotation = Transform.normalizeRadian(
            parentRadian + (currentRadian - parentRadian) * this._weight
        );
        parentGlobal.toMatrix(parent.globalTransformMatrix);

        global.rotation = Transform.normalizeRadian(
            currentRadian + radianA
        );
        global.toMatrix(globalTransformMatrix);
    }

    /**
     * 强制更新
     */
    public invalidUpdate(): void {
        // 标记所有相关骨骼为需要更新
        this._root.invalidUpdate();
        if (this._bone) {
            this._bone.invalidUpdate();
        }
    }
}
```

### 3.3 骨骼权重系统

```typescript
/**
 * 骨骼权重系统 - 管理顶点权重和蒙皮变形
 *
 * 职责：
 * - 计算顶点权重
 * - 执行蒙皮变形
 * - 优化权重分布
 */
class VertexWeightSystem {
    private _weightCache = new Map<string, WeightData>();
    private _deformCache = new Map<string, DeformData>();

    /**
     * 计算顶点权重
     */
    public calculateVertexWeights(
        vertices: number[],
        bones: Bone[],
        options: WeightOptions = {}
    ): WeightData {
        const cacheKey = this.generateWeightKey(vertices, bones);

        if (this._weightCache.has(cacheKey)) {
            return this._weightCache.get(cacheKey)!;
        }

        const weightData = new WeightData();
        const vertexCount = vertices.length / 2;

        // 为每个顶点计算权重
        for (let i = 0; i < vertexCount; i++) {
            const vertex = {
                x: vertices[i * 2],
                y: vertices[i * 2 + 1]
            };

            // 计算到每个骨骼的距离
            const distances = this.calculateDistances(vertex, bones);

            // 根据距离计算权重
            const weights = this.calculateWeightsFromDistances(distances, options);

            // 存储权重数据
            weightData.vertexWeights.push(weights);
            weightData.boneIndices.push(this.getInfluentialBones(weights));
        }

        this._weightCache.set(cacheKey, weightData);
        return weightData;
    }

    /**
     * 计算距离
     */
    private calculateDistances(vertex: { x: number; y: number }, bones: Bone[]): number[] {
        return bones.map(bone => {
            const dx = vertex.x - bone.global.x;
            const dy = vertex.y - bone.global.y;
            return Math.sqrt(dx * dx + dy * dy);
        });
    }

    /**
     * 根据距离计算权重
     */
    private calculateWeightsFromDistances(
        distances: number[],
        options: WeightOptions
    ): number[] {
        const weights: number[] = [];
        const falloff = options.falloff || 2.0;
        const maxInfluence = options.maxInfluence || 4;

        // 1. 计算距离的倒数作为初始权重
        let totalWeight = 0;
        for (const distance of distances) {
            const weight = Math.pow(1 / (distance + 1), falloff);
            weights.push(weight);
            totalWeight += weight;
        }

        // 2. 归一化权重
        if (totalWeight > 0) {
            for (let i = 0; i < weights.length; i++) {
                weights[i] /= totalWeight;
            }
        }

        // 3. 限制最大影响骨骼数量
        if (weights.length > maxInfluence) {
            this.limitInfluenceCount(weights, maxInfluence);
        }

        return weights;
    }

    /**
     * 限制影响骨骼数量
     */
    private limitInfluenceCount(weights: number[], maxInfluence: number): void {
        // 1. 找到权重最大的骨骼
        const sortedIndices = weights
            .map((weight, index) => ({ weight, index }))
            .sort((a, b) => b.weight - a.weight)
            .slice(0, maxInfluence)
            .map(item => item.index);

        // 2. 保留最大的权重，其他设为0
        for (let i = 0; i < weights.length; i++) {
            if (!sortedIndices.includes(i)) {
                weights[i] = 0;
            }
        }

        // 3. 重新归一化
        const totalWeight = sortedIndices.reduce((sum, index) => sum + weights[index], 0);
        if (totalWeight > 0) {
            for (const index of sortedIndices) {
                weights[index] /= totalWeight;
            }
        }
    }

    /**
     * 获取有影响力的骨骼
     */
    private getInfluentialBones(weights: number[]): number[] {
        const influentialBones: number[] = [];
        const threshold = 0.01; // 权重阈值

        for (let i = 0; i < weights.length; i++) {
            if (weights[i] > threshold) {
                influentialBones.push(i);
            }
        }

        return influentialBones;
    }

    /**
     * 执行蒙皮变形
     */
    public deformSkin(
        vertices: number[],
        bones: Bone[],
        weightData: WeightData
    ): number[] {
        const deformedVertices = [...vertices];
        const vertexCount = vertices.length / 2;

        for (let i = 0; i < vertexCount; i++) {
            const originalPos = {
                x: vertices[i * 2],
                y: vertices[i * 2 + 1]
            };

            // 计算变形后的位置
            const deformedPos = this.calculateDeformedPosition(
                originalPos,
                bones,
                weightData.vertexWeights[i]
            );

            deformedVertices[i * 2] = deformedPos.x;
            deformedVertices[i * 2 + 1] = deformedPos.y;
        }

        return deformedVertices;
    }

    /**
     * 计算变形后的位置
     */
    private calculateDeformedPosition(
        vertex: { x: number; y: number },
        bones: Bone[],
        weights: number[]
    ): { x: number; y: number } {
        let finalX = 0;
        let finalY = 0;
        let totalWeight = 0;

        for (let i = 0; i < bones.length && i < weights.length; i++) {
            const weight = weights[i];
            if (weight > 0) {
                const bone = bones[i];
                const transformedPos = bone.globalTransformMatrix.transformPoint(vertex.x, vertex.y);

                finalX += transformedPos.x * weight;
                finalY += transformedPos.y * weight;
                totalWeight += weight;
            }
        }

        // 归一化
        if (totalWeight > 0) {
            finalX /= totalWeight;
            finalY /= totalWeight;
        }

        return { x: finalX, y: finalY };
    }
}
```

## 4. 关键算法

### 4.1 IK求解算法

```typescript
/**
 * IK求解算法
 */
class IKSolverAlgorithms {
    /**
     * CCD (Cyclic Coordinate Descent) 算法
     */
    static solveCCD(
        bones: Bone[],
        target: { x: number; y: number },
        tolerance: number = 0.1,
        maxIterations: number = 10
    ): boolean {
        const endEffector = bones[bones.length - 1];

        for (let iteration = 0; iteration < maxIterations; iteration++) {
            // 从末端开始向前迭代
            for (let i = bones.length - 1; i >= 0; i--) {
                const bone = bones[i];
                const currentPos = { x: bone.global.x, y: bone.global.y };

                // 计算当前骨骼到目标的向量
                const toTarget = {
                    x: target.x - currentPos.x,
                    y: target.y - currentPos.y
                };

                // 计算当前骨骼到末端效应器的向量
                const toEndEffector = {
                    x: endEffector.global.x - currentPos.x,
                    y: endEffector.global.y - currentPos.y
                };

                // 计算旋转角度
                const angleToTarget = Math.atan2(toTarget.y, toTarget.x);
                const angleToEnd = Math.atan2(toEndEffector.y, toEndEffector.x);
                const rotationAngle = angleToTarget - angleToEnd;

                // 应用旋转
                bone.global.rotation += rotationAngle;
                bone.global.toMatrix(bone.globalTransformMatrix);

                // 更新后续骨骼
                this.updateChildBones(bone);
            }

            // 检查是否达到目标
            const distance = Math.sqrt(
                (endEffector.global.x - target.x) ** 2 +
                (endEffector.global.y - target.y) ** 2
            );

            if (distance <= tolerance) {
                return true; // 成功
            }
        }

        return false; // 失败
    }

    /**
     * FABRIK (Forward And Backward Reaching Inverse Kinematics) 算法
     */
    static solveFABRIK(
        bones: Bone[],
        target: { x: number; y: number },
        tolerance: number = 0.1
    ): boolean {
        const positions: { x: number; y: number }[] = [];
        const lengths: number[] = [];

        // 1. 初始化位置和长度
        for (let i = 0; i < bones.length; i++) {
            positions.push({ x: bones[i].global.x, y: bones[i].global.y });
            if (i > 0) {
                const dx = positions[i].x - positions[i-1].x;
                const dy = positions[i].y - positions[i-1].y;
                lengths.push(Math.sqrt(dx * dx + dy * dy));
            }
        }

        // 2. 前向传递
        positions[0] = { x: bones[0].global.x, y: bones[0].global.y };
        for (let i = 1; i < positions.length; i++) {
            const direction = {
                x: (positions[i].x - positions[i-1].x) / lengths[i-1],
                y: (positions[i].y - positions[i-1].y) / lengths[i-1]
            };

            positions[i] = {
                x: positions[i-1].x + direction.x * lengths[i-1],
                y: positions[i-1].y + direction.y * lengths[i-1]
            };
        }

        // 3. 后向传递
        positions[positions.length - 1] = target;
        for (let i = positions.length - 2; i >= 0; i--) {
            const direction = {
                x: (positions[i].x - positions[i+1].x) / lengths[i],
                y: (positions[i].y - positions[i+1].y) / lengths[i]
            };

            positions[i] = {
                x: positions[i+1].x + direction.x * lengths[i],
                y: positions[i+1].y + direction.y * lengths[i]
            };
        }

        // 4. 更新骨骼位置
        for (let i = 0; i < bones.length; i++) {
            bones[i].global.x = positions[i].x;
            bones[i].global.y = positions[i].y;
            bones[i].global.toMatrix(bones[i].globalTransformMatrix);
        }

        return true;
    }

    /**
     * 更新子骨骼
     */
    private static updateChildBones(bone: Bone): void {
        // 递归更新所有子骨骼
        for (const child of bone.children) {
            child.update(true);
            this.updateChildBones(child);
        }
    }
}
```

```typescript
/**
 * 物理约束算法
 */
class PhysicsConstraintAlgorithms {
    /**
     * 重力约束
     */
    static applyGravity(bones: Bone[], gravity: { x: number; y: number }): void {
        for (const bone of bones) {
            // 应用重力加速度
            bone.velocity.x += gravity.x * Time.deltaTime;
            bone.velocity.y += gravity.y * Time.deltaTime;

            // 更新位置
            bone.global.x += bone.velocity.x * Time.deltaTime;
            bone.global.y += bone.velocity.y * Time.deltaTime;
        }
    }

    /**
     * 风力约束
     */
    static applyWindForce(
        bones: Bone[],
        wind: { x: number; y: number; turbulence: number }
    ): void {
        for (const bone of bones) {
            // 添加随机扰动
            const turbulence = {
                x: (Math.random() - 0.5) * wind.turbulence,
                y: (Math.random() - 0.5) * wind.turbulence
            };

            const force = {
                x: wind.x + turbulence.x,
                y: wind.y + turbulence.y
            };

            // 根据骨骼质量应用力
            const acceleration = {
                x: force.x / bone.mass,
                y: force.y / bone.mass
            };

            bone.velocity.x += acceleration.x * Time.deltaTime;
            bone.velocity.y += acceleration.y * Time.deltaTime;
        }
    }

    /**
     * 弹性约束
     */
    static applySpringConstraint(
        bone: Bone,
        anchor: { x: number; y: number },
        stiffness: number,
        damping: number
    ): void {
        // 计算到锚点的距离
        const dx = bone.global.x - anchor.x;
        const dy = bone.global.y - anchor.y;
        const distance = Math.sqrt(dx * dx + dy * dy);

        if (distance > 0) {
            // 弹性力
            const springForce = -stiffness * distance;
            const forceDirection = { x: dx / distance, y: dy / distance };

            // 阻尼力
            const dampingForce = -damping * (bone.velocity.x * forceDirection.x + bone.velocity.y * forceDirection.y);

            // 合力
            const totalForce = springForce + dampingForce;

            // 应用力
            const acceleration = {
                x: totalForce * forceDirection.x / bone.mass,
                y: totalForce * forceDirection.y / bone.mass
            };

            bone.velocity.x += acceleration.x * Time.deltaTime;
            bone.velocity.y += acceleration.y * Time.deltaTime;
        }
    }
}
```

```typescript
     * 批量求解约束
     */
    static batchSolve(constraints: Constraint[]): void {
        // 1. 拓扑排序约束
        const sortedConstraints = this.topologicalSort(constraints);

        // 2. 并行求解独立约束
        const independentGroups = this.groupIndependentConstraints(sortedConstraints);

        for (const group of independentGroups) {
            // 并行求解
            group.forEach(constraint => constraint.update());
        }
    }

    /**
     * 拓扑排序
     */
    private static topologicalSort(constraints: Constraint[]): Constraint[] {
        const visited = new Set<Constraint>();
        const result: Constraint[] = [];

        function visit(constraint: Constraint): void {
            if (visited.has(constraint)) return;
            visited.add(constraint);

            // 先访问依赖的约束
            const dependencies = this.getDependencies(constraint);
            for (const dep of dependencies) {
                visit(dep);
            }

            result.push(constraint);
        }

        for (const constraint of constraints) {
            visit(constraint);
        }

        return result;
    }

    /**
     * 分组独立约束
     */
    private static groupIndependentConstraints(constraints: Constraint[]): Constraint[][] {
        const groups: Constraint[][] = [];
        const processed = new Set<Constraint>();

        for (const constraint of constraints) {
            if (processed.has(constraint)) continue;

            const group: Constraint[] = [constraint];
            processed.add(constraint);

            // 查找可以同时求解的约束
            for (const other of constraints) {
                if (processed.has(other)) continue;

                if (this.areIndependent(constraint, other)) {
                    group.push(other);
                    processed.add(other);
                }
            }

            groups.push(group);
        }

        return groups;
    }

    /**
     * 检查约束是否独立
     */
    private static areIndependent(c1: Constraint, c2: Constraint): boolean {
        // 检查骨骼是否重叠
        const bones1 = this.getInvolvedBones(c1);
        const bones2 = this.getInvolvedBones(c2);

        for (const bone1 of bones1) {
            for (const bone2 of bones2) {
                if (bone1 === bone2) {
                    return false;
                }
            }
        }

        return true;
    }

    /**
     * 获取约束涉及的骨骼
     */
    private static getInvolvedBones(constraint: Constraint): Bone[] {
        const bones: Bone[] = [];

        if (constraint._root) bones.push(constraint._root);
        if (constraint._target) bones.push(constraint._target);
        if (constraint._bone) bones.push(constraint._bone);

        return bones;
    }
}
```

### 5.2 权重计算优化

```typescript
/**
 * 权重计算性能优化
 */
class WeightCalculationOptimizer {
    private static _spatialIndex: SpatialHashGrid;
    private static _weightCache = new Map<string, WeightData>();

    /**
     * 使用空间索引优化权重计算
     */
    static calculateWeightsOptimized(
        vertices: number[],
        bones: Bone[],
        options: WeightOptions
    ): WeightData {
        // 1. 构建骨骼空间索引
        if (!this._spatialIndex) {
            this._spatialIndex = new SpatialHashGrid(50); // 网格大小50
        }

        this._spatialIndex.clear();
        for (let i = 0; i < bones.length; i++) {
            const bone = bones[i];
            this._spatialIndex.insert(bone, bone.global.x, bone.global.y);
        }

        // 2. 为每个顶点计算权重
        const weightData = new WeightData();
        const vertexCount = vertices.length / 2;

        for (let i = 0; i < vertexCount; i++) {
            const vertex = {
                x: vertices[i * 2],
                y: vertices[i * 2 + 1]
            };

            // 使用空间索引查找附近的骨骼
            const nearbyBones = this._spatialIndex.queryRange(
                vertex.x - options.searchRadius,
                vertex.y - options.searchRadius,
                vertex.x + options.searchRadius,
                vertex.y + options.searchRadius
            );

            // 只计算附近骨骼的权重
            const weights = this.calculateWeightsForNearbyBones(vertex, nearbyBones);

            weightData.vertexWeights.push(weights);
            weightData.boneIndices.push(this.getInfluentialBones(weights));
        }

        return weightData;
    }

    /**
     * 空间哈希网格
     */
    private static class SpatialHashGrid {
        private grid: Map<string, Bone[]>;
        private cellSize: number;

        constructor(cellSize: number) {
            this.cellSize = cellSize;
            this.grid = new Map();
        }

        insert(bone: Bone, x: number, y: number): void {
            const cellX = Math.floor(x / this.cellSize);
            const cellY = Math.floor(y / this.cellSize);
            const key = `${cellX},${cellY}`;

            if (!this.grid.has(key)) {
                this.grid.set(key, []);
            }

            this.grid.get(key)!.push(bone);
        }

        queryRange(minX: number, minY: number, maxX: number, maxY: number): Bone[] {
            const result: Bone[] = [];
            const minCellX = Math.floor(minX / this.cellSize);
            const minCellY = Math.floor(minY / this.cellSize);
            const maxCellX = Math.floor(maxX / this.cellSize);
            const maxCellY = Math.floor(maxY / this.cellSize);

            for (let x = minCellX; x <= maxCellX; x++) {
                for (let y = minCellY; y <= maxCellY; y++) {
                    const key = `${x},${y}`;
                    const bones = this.grid.get(key);
                    if (bones) {
                        result.push(...bones);
                    }
                }
            }

            return result;
        }

        clear(): void {
            this.grid.clear();
        }
    }
}
```

## 8. 最佳实践

### 8.1 约束设计原则

1. **合理的约束层级** - 避免循环依赖
2. **适当的约束权重** - 平衡多个约束的影响
3. **优化求解顺序** - 先求解基础约束
4. **限制约束复杂度** - 避免过多的约束链

### 8.2 性能优化建议

1. **使用空间索引** - 优化权重计算
2. **批量处理约束** - 减少更新开销
3. **缓存计算结果** - 避免重复计算
4. **异步处理** - 复杂计算使用Web Worker
